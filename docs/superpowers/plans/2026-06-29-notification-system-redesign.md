# Notification System Redesign Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace scattered buyer-only inbox writes with a transaction-safe, typed, three-audience notification system while preserving the current buyer App `/inbox` experience during migration.

**Architecture:** Add `NotificationModule` with typed events, a route/template registry, `NotificationOutbox`, and `NotificationMessage`. Existing business modules emit notification events into the outbox inside their normal transaction boundary; a dispatcher creates recipient messages after commit. Buyer App reads the new message table through the existing `/inbox` API first, while seller/admin notification APIs and UI are added after the buyer path is stable.

**Tech Stack:** NestJS, Prisma, PostgreSQL, React Native Expo, Vite React, Ant Design, TanStack Query, Jest.

**Post-review correction (2026-06-30):** Seller notification recipient keys must use `seller:{User.id}`, not `seller:{CompanyStaff.id}` from seller JWT `sub`; seller controllers therefore read `CurrentUser.userId`. Admin notifications use `admin:{AdminUser.id}` and registry falls back to ACTIVE admins when an admin-audience event omits explicit `adminUserIds`. Dispatcher must reclaim stale `PROCESSING` outbox rows.

## Global Constraints

- First implementation phase must not connect real Push, SMS, Enterprise WeChat, or marketing broadcast delivery.
- Buyer messages must not contain raw phone numbers, ID numbers, bank cards, or full detailed addresses.
- Seller notifications must respect buyer public number and existing seller privacy boundaries.
- Notification actions must use registered `routeKey` values, not arbitrary frontend route strings.
- Repeated payment callbacks, logistics callbacks, and cron retries must not create duplicate recipient messages.
- Existing buyer `/inbox` endpoints must remain compatible while the App migrates to the new data source.
- Seller/admin notifications must have backend model and APIs even if the first visible UI remains minimal.

```ts
import { Injectable } from '@nestjs/common';
import {
  NotificationEvent,
  NotificationMessageDraft,
  NotificationResolveResult,
} from './notification.types';

@Injectable()
export class NotificationRegistry {
  resolve(event: NotificationEvent): NotificationResolveResult {
    if (event.eventType === 'order.shipped') {
      const buyerUserId = String(event.payload.buyerUserId || '');
      const orderId = String(event.payload.orderId || event.aggregateId);
      return {
        messages: [
          this.buildMessage(event, {
            recipientKind: 'BUYER_USER',
            recipientKey: `buyer:${buyerUserId}`,
            audience: 'BUYER_APP',
            category: 'order',
            title: '订单已发货',
            body: '您的订单已发货，可查看物流进度。',
            severity: 'SUCCESS',
            entityType: 'order',
            entityId: orderId,
            action: { routeKey: 'ORDER_DETAIL', params: { id: orderId } },
          }),
        ],
      };
    }

    throw new Error(`未注册的通知事件: ${event.eventType}`);
  }

  private buildMessage(
    event: NotificationEvent,
    input: Omit<NotificationMessageDraft, 'eventType' | 'idempotencyKey'>,
  ): NotificationMessageDraft {
    return {
      ...input,
      eventType: event.eventType,
      idempotencyKey: `${event.idempotencyKey || `${event.eventType}:${event.aggregateType}:${event.aggregateId}`}:${input.recipientKey}`,
    };
  }
}
```

import { NotificationEvent } from './notification.types';

type PrismaLike = Pick<PrismaService, 'notificationOutbox'>;

@Injectable()
export class NotificationService {
  constructor(private prisma: PrismaService) {}

  async emit(event: NotificationEvent, client: PrismaLike = this.prisma) {
    const idempotencyKey =
      event.idempotencyKey || `${event.eventType}:${event.aggregateType}:${event.aggregateId}`;

    return client.notificationOutbox.upsert({
      where: { idempotencyKey },
      update: {},
      create: {
        eventType: event.eventType,
        aggregateType: event.aggregateType,
        aggregateId: event.aggregateId,
        idempotencyKey,
        payload: event as any,
      },
    });
  }
}

```ts
import { Injectable, Logger } from '@nestjs/common';
import { Cron, CronExpression } from '@nestjs/schedule';
import { PrismaService } from '../../prisma/prisma.service';
import { NotificationRegistry } from './notification.registry';
import { NotificationEvent } from './notification.types';

@Injectable()
export class NotificationDispatcherService {
  private readonly logger = new Logger(NotificationDispatcherService.name);

  constructor(
    private prisma: PrismaService,
    private registry: NotificationRegistry,
  ) {}

  @Cron(CronExpression.EVERY_10_SECONDS)
  async dispatchCron() {
    await this.dispatchPending(50);
  }

  async dispatchPending(limit = 50) {
    const rows = await this.prisma.notificationOutbox.findMany({
      where: { status: 'PENDING', runAt: { lte: new Date() } },
      orderBy: { createdAt: 'asc' },
      take: limit,
    });

    for (const row of rows) {
      await this.dispatchOne(row);
    }
  }

  private async dispatchOne(row: any) {
    try {
      await this.prisma.notificationOutbox.update({
        where: { id: row.id },
        data: { status: 'PROCESSING', processingAt: new Date(), attempts: { increment: 1 } },
      });
      const event = row.payload as NotificationEvent;
      const resolved = this.registry.resolve(event);
      for (const message of resolved.messages) {
        await this.prisma.notificationMessage.upsert({
          where: {
            recipientKey_idempotencyKey: {
              recipientKey: message.recipientKey,
              idempotencyKey: message.idempotencyKey,
            },
          },
          update: {},
          create: message as any,
        });
      }
      await this.prisma.notificationOutbox.update({
        where: { id: row.id },
        data: { status: 'SENT', processedAt: new Date(), lastError: null },
      });
    } catch (error: any) {
      this.logger.warn(`通知派发失败: outboxId=${row.id}, error=${error?.message || error}`);
      await this.prisma.notificationOutbox.update({
        where: { id: row.id },
        data: {
          status: row.attempts >= 4 ? 'FAILED' : 'PENDING',
          lastError: String(error?.message || error).slice(0, 1000),
          runAt: new Date(Date.now() + Math.min(60_000, 2 ** Math.max(row.attempts, 1) * 1000)),
        },
      });
    }
  }
}
```

import { Injectable, NotFoundException } from '@nestjs/common';
import { PrismaService } from '../../prisma/prisma.service';

@Injectable()
export class NotificationMessageService {
  constructor(private prisma: PrismaService) {}

  async list(recipientKey: string, category?: string, unreadOnly?: boolean, page = 1, pageSize = 20) {
    const take = Math.min(Math.max(pageSize, 1), 50);
    const where: any = { recipientKey };
    if (category) where.category = category;
    if (unreadOnly) where.readAt = null;
    const rows = await this.prisma.notificationMessage.findMany({
      where,
      orderBy: { createdAt: 'desc' },
      skip: (Math.max(page, 1) - 1) * take,
      take,
    });
    return rows.map((row) => this.map(row));
  }

  async unreadCount(recipientKey: string) {
    return this.prisma.notificationMessage.count({ where: { recipientKey, readAt: null } });
  }

  async markRead(recipientKey: string, id: string) {
    const row = await this.prisma.notificationMessage.findUnique({ where: { id } });
    if (!row || row.recipientKey !== recipientKey) throw new NotFoundException('消息不存在');
    await this.prisma.notificationMessage.update({
      where: { id },
      data: { readAt: row.readAt || new Date() },
    });
    return this.list(recipientKey);
  }

  async markAllRead(recipientKey: string) {
    await this.prisma.notificationMessage.updateMany({
      where: { recipientKey, readAt: null },
      data: { readAt: new Date() },
    });
    return this.list(recipientKey);
  }

  private map(row: any) {
    return {
      id: row.id,
      category: row.category,
      type: row.eventType,
      title: row.title,
      content: row.body,
      createdAt: row.createdAt instanceof Date ? row.createdAt.toISOString() : row.createdAt,
      unread: !row.readAt,
      action: row.action || undefined,
      target: row.action || undefined,
    };
  }
}

type BuyerAction = {
  routeKey: string;
  params?: Record<string, string>;
};

const ROUTES: Record<string, string> = {
  ORDER_DETAIL: '/orders/[id]',
  ORDER_TRACK: '/orders/track',
  AFTER_SALE_DETAIL: '/orders/after-sale-detail/[id]',
  INVOICE_DETAIL: '/invoices/[id]',
  WALLET: '/me/wallet',
  COUPONS: '/me/coupons',
  DIGITAL_ASSETS: '/me/digital-assets',
  GROUP_BUY_DETAIL: '/group-buy/[activityId]',
  CS_SESSION: '/cs',
};

export function resolveBuyerNotificationRoute(action?: BuyerAction | null) {
  if (!action?.routeKey) return null;
  const pathname = ROUTES[action.routeKey];
  if (!pathname) return null;
  return { pathname, params: action.params };

```ts
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

const legacyRouteToAction = (target: any) => {
  const route = target?.route;
  if (route?.startsWith('/orders/')) return { routeKey: 'ORDER_DETAIL', params: { id: route.split('/').pop() } };
  if (route === '/me/coupons') return { routeKey: 'COUPONS' };
  if (route === '/me/wallet') return { routeKey: 'WALLET' };
  return undefined;
};

async function main() {
  const rows = await prisma.inboxMessage.findMany({ orderBy: { createdAt: 'asc' } });
  for (const row of rows) {
    await prisma.notificationMessage.upsert({
      where: {
        recipientKey_idempotencyKey: {
          recipientKey: `buyer:${row.userId}`,
          idempotencyKey: `legacy-inbox:${row.id}`,
        },
      },
      update: {},
      create: {
        recipientKind: 'BUYER_USER',
        recipientKey: `buyer:${row.userId}`,
        audience: 'BUYER_APP',
        category: ['interaction', 'transaction', 'system'].includes(row.category) ? row.category : 'system',
        eventType: row.type || 'system.legacyInfo',
        title: row.title,
        body: row.content,
        severity: 'INFO',
        entityType: 'legacyInbox',
        entityId: row.id,
        action: legacyRouteToAction(row.target) as any,
        idempotencyKey: `legacy-inbox:${row.id}`,
        readAt: row.unread ? null : row.updatedAt,
        createdAt: row.createdAt,
      },
    });
  }
}

main().finally(() => prisma.$disconnect());
```
